    <script>
        // Demo Toast functions
        function showToast(title, message, type = 'info') {
            const toastContainer = document.getElementById('toastContainer') || createToastContainer();
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;

            toast.innerHTML = `
                <div class="toast-header">
                    <h4 class="toast-title">${title}</h4>
                    <button class="toast-close" onclick="removeToast(this.parentElement.parentElement)">×</button>
                </div>
                <div class="toast-body">${message}</div>
            `;

            toastContainer.appendChild(toast);

            setTimeout(() => {
                toast.classList.add('show');
            }, 100);

            setTimeout(() => {
                removeToast(toast);
            }, 8000);
        }

        function createToastContainer() {
            const container = document.createElement('div');
            container.id = 'toastContainer';
            container.className = 'toast-container';
            document.body.appendChild(container);
            return container;
        }

        function removeToast(toast) {
            if (toast && toast.parentElement) {
                toast.classList.add('fade-out');
                setTimeout(() => {
                    if (toast.parentElement) {
                        toast.remove();
                    }
                }, 400);
            }
        }

        // Period selector functionality
        document.querySelectorAll('.period-btn').forEach(btn => {
            btn.addEventListener('click', function() {
                document.querySelectorAll('.period-btn').forEach(b => b.classList.remove('active'));
                this.classList.add('active');
                showToast('Zeitraum geändert', `Analytics für: ${this.textContent}`, 'info');
            });
        });

        // Interactive chart bars
        document.querySelectorAll('.chart-bar').forEach(bar => {
            bar.addEventListener('click', function() {
                const title = this.getAttribute('title');
                showToast('Produktivitätsdetails', title || 'Details für diesen Tag', 'info');
            });
        });

        // Time slot interactions
        document.querySelectorAll('.time-slot').forEach(slot => {
            slot.addEventListener('click', function() {
                const hour = this.textContent;
                const activity = this.classList.contains('high-activity') ? 'Hoch' :
                               this.classList.contains('medium-activity') ? 'Mittel' : 'Niedrig';
                showToast('Stundendetails', `${hour}:00 Uhr - Aktivität: ${activity}`, 'info');
            });
        });

        // Demo notification on load
        setTimeout(() => {
            showToast('Analytics Mockup', 'Detaillierte Produktivitätsanalyse mit interaktiven Charts und KI-Empfehlungen', 'info');
        }, 1000);
    </script>
