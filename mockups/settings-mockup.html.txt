    <script>
        // Demo Toast functions
        function showToast(title, message, type = 'info') {
            const toastContainer = document.getElementById('toastContainer') || createToastContainer();
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;

            toast.innerHTML = `
                <div class="toast-header">
                    <h4 class="toast-title">${title}</h4>
                    <button class="toast-close" onclick="removeToast(this.parentElement.parentElement)">√ó</button>
                </div>
                <div class="toast-body">${message}</div>
            `;

            toastContainer.appendChild(toast);

            setTimeout(() => {
                toast.classList.add('show');
            }, 100);

            setTimeout(() => {
                removeToast(toast);
            }, 8000);
        }

        function createToastContainer() {
            const container = document.createElement('div');
            container.id = 'toastContainer';
            container.className = 'toast-container';
            document.body.appendChild(container);
            return container;
        }

        function removeToast(toast) {
            if (toast && toast.parentElement) {
                toast.classList.add('fade-out');
                setTimeout(() => {
                    if (toast.parentElement) {
                        toast.remove();
                    }
                }, 400);
            }
        }

        // Toggle switch functionality
        function toggleSwitch(element) {
            element.classList.toggle('active');
            const isActive = element.classList.contains('active');
            const label = element.parentElement.parentElement.querySelector('.setting-label').textContent;
            showToast('Einstellung ge√§ndert', `${label}: ${isActive ? 'Aktiviert' : 'Deaktiviert'}`, 'info');
        }

        // Navigation functionality
        document.querySelectorAll('.settings-nav-item').forEach(item => {
            item.addEventListener('click', function(e) {
                e.preventDefault();

                // Update active state
                document.querySelectorAll('.settings-nav-item').forEach(nav => nav.classList.remove('active'));
                this.classList.add('active');

                // Show toast
                const sectionName = this.textContent.trim();
                showToast('Navigation', `Zu Abschnitt gewechselt: ${sectionName}`, 'info');
            });
        });

        // Demo notification on load
        setTimeout(() => {
            showToast('Settings Mockup', 'Umfassende Systemkonfiguration mit allen verf√ºgbaren Optionen', 'info');
        }, 1000);

        // Form interactions
        document.querySelectorAll('.form-select, .form-input').forEach(input => {
            input.addEventListener('change', function() {
                const label = this.previousElementSibling?.textContent || 'Einstellung';
                showToast('√Ñnderung gespeichert', `${label} wurde aktualisiert`, 'success');
            });
        });
    </script>
